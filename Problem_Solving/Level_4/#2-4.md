//#2-4
//program to know Is leap year or not
#include <iostream>
#include <string>

using namespace std;

bool IsLeapYear(short Year)
{
	if (Year % 400 == 0)
		return true;
	if (Year % 100 == 0)
		return false;
	if (Year % 4 == 0)
		return true;
	else
		return false;
}

short ReadYear()
{
	short Year;

	cout << "Enter the year? \n";
	cin >> Year;

	return Year;
}
int main()
{
	short Year = ReadYear();
	
	if (IsLeapYear(Year))
		cout << "\nYes, Year [" << Year << "] is a leap year.\n";
	else
		cout << "\nNo, Year [" << Year << "] is NOT a leap year.\n";
	
	system("pause > 0");
	return 0;
}
