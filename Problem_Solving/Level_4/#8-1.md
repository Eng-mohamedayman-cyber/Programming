#include <iostream>
#include <string>

using namespace std;

short ReadYear()
{
	short Year;

	cout << "\nPlease enter a year? ";
	cin >> Year;

	return Year;
}
short ReadMonth()
{
	short Month;

	cout << "\nPlease enter a Month? ";
	cin >> Month;

	return Month;
}
string DayShortName(short DayOfWeekOrder)
{
	string arrDayName[] = { "Sun","Mon","Tue","Wen","Thur","Fri","Sat" };

	return arrDayName[DayOfWeekOrder];
}
short DayOfWeekOrder(short Day, short Month, short Year)
{
	int a = (14 - Month) / 12;
	int y = Year - a;
	int m = Month + 12 * a - 2;
	int d = (Day + y + (y / 4) - (y / 100) + (y / 400) + ((31 * m) / 12)) % 7;

	return d;
}
string MonthShortName(short Month)
{
	string arrMonthName[] = { "jan","feb","mar","apr","may","jun","jul","aug","sep","oct","nov","dec" };

	return arrMonthName[Month - 1];
}
bool IsLeapYear(short Year)
{
	return (Year % 4 == 0 && Year % 100 != 0) || (Year % 400 == 0);
}
short NumberOfDaysInAMonth(short Month, short Year)
{
	if (Month < 1 || Month > 12)
		return 0;

	int NumberOfDays[13] = { 0,31,28,31,30,31,30,31,31,30,31,30,31 };
	return (Month == 2) ? ((IsLeapYear(Year)) ? 29 : 28) : NumberOfDays[Month];
}
void PrintMonthCalendar(short Month, short Year) {

	
	short NumberOfDays = NumberOfDaysInAMonth(Month, Year);
	short current = DayOfWeekOrder(1, Month, Year);

//	printf("\n _______________%s_______________\n\n",MonthShortName(Month).c_str());
	
	cout << "\n_________________" << MonthShortName(Month) << "_________________\n" << endl;
	cout << "  Sun  Mon  Tue  Wen  Thur  Fri  Sat\n" << endl;
	int i;
	for (i = 0; i < current; i++)
		printf("     ");
	for (int j = 1; j <= NumberOfDays; j++)
	{
		printf("%5d", j);

		if (++i == 7)
		{
			i = 0;
			printf("\n");
		}
	}
	
	cout << "\n_____________________________________" << endl;

}
int main()
{
	short Year = ReadYear();
	short Month = ReadMonth();

	PrintMonthCalendar(Month,Year);

	system("pause > 0");
	return 0;
}
